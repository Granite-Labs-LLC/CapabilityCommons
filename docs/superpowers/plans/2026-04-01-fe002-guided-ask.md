- [ ] **Step 1: Replace the entire AskTutor.tsx file**

Replace the contents of `apps/site/src/islands/AskTutor.tsx` with:

```tsx
import { useState, useCallback } from 'react';
import type {
  GraphNode,
  RetrievalIntent,
  AskResponse,
  ImplementationStep,
  SafetyBlock,
  AskCitation,
  RelatedObject,
} from '../lib/types';

const INTENTS: { id: RetrievalIntent; label: string; desc: string }[] = [
  { id: 'how_to', label: 'How to do something', desc: 'Step-by-step guidance for a practical task' },
  { id: 'learn_path', label: 'What to learn next', desc: 'Recommended sequence from where you are' },
  { id: 'why', label: 'Why does this matter?', desc: 'Understand the reasoning behind a capability' },
  { id: 'compare_options', label: 'Compare options', desc: 'Trade-offs between approaches' },
  { id: 'localize', label: 'Adapt for my situation', desc: 'Context-specific guidance (climate, budget, housing)' },
  { id: 'debug_failure', label: 'Debug a problem', desc: 'Something went wrong — help me fix it' },
  { id: 'teach_forward', label: 'How to teach this', desc: 'Prepare to teach someone else' },
  { id: 'safety_check', label: 'Safety check', desc: 'Verify whether something is safe to do' },
];

const CONTEXTS = {
  housing: ['Renter', 'Homeowner', 'Apartment', 'House', 'Mobile home'],
  climate: ['Cold', 'Temperate', 'Hot/humid', 'Arid'],
  budget: ['Minimal ($0-25)', 'Low ($25-100)', 'Medium ($100-500)', 'Flexible'],
  experience: ['Complete beginner', 'Some experience', 'Comfortable with tools', 'Experienced'],
};

const CONTEXT_TO_ASK_FIELD: Record<string, string> = {
  housing: 'housing_type',
  climate: 'climate_zone',
  budget: 'budget_profile',
  experience: 'experience_level',
};

interface Props {
  nodes: GraphNode[];
  apiBase: string;
}

interface Message {
  role: 'user' | 'assistant';
  query?: string;
  intent?: string;
  response?: AskResponse;
}

export default function AskTutor({ nodes, apiBase }: Props) {
  const [query, setQuery] = useState('');
  const [selectedIntent, setSelectedIntent] = useState<RetrievalIntent | null>(null);
  const [context, setContext] = useState<Record<string, string>>({});
  const [showContext, setShowContext] = useState(false);
  const [messages, setMessages] = useState<Message[]>([]);
  const [loading, setLoading] = useState(false);
  const [conversationId, setConversationId] = useState<string | null>(null);

  const handleSubmit = useCallback(async () => {
    if (!query.trim()) return;

    const userMsg: Message = { role: 'user', query, intent: selectedIntent || undefined };
    setMessages((prev) => [...prev, userMsg]);
    setLoading(true);

    const askContext: Record<string, string> = {};
    for (const [key, value] of Object.entries(context)) {
      const field = CONTEXT_TO_ASK_FIELD[key];
      if (field) askContext[field] = value;
    }

    try {
      const res = await fetch(`${apiBase}/v1/public/ask`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          query,
          intent: selectedIntent || undefined,
          context: Object.keys(askContext).length > 0 ? askContext : undefined,
          conversation_id: conversationId || undefined,
        }),
      });

      if (res.ok) {
        const askResponse: AskResponse = await res.json();
        if (askResponse.conversation_id) setConversationId(askResponse.conversation_id);
        setMessages((prev) => [...prev, { role: 'assistant', response: askResponse }]);
      } else {
        const mockResponse = generateMockResponse(query, selectedIntent || 'how_to', nodes);
        setMessages((prev) => [...prev, { role: 'assistant', response: mockResponse }]);
      }
    } catch {
      const mockResponse = generateMockResponse(query, selectedIntent || 'how_to', nodes);
      setMessages((prev) => [...prev, { role: 'assistant', response: mockResponse }]);
    }

    setQuery('');
    setLoading(false);
  }, [query, selectedIntent, context, nodes, apiBase, conversationId]);

  return (
    <div className="ask-tutor">
      {messages.length > 0 && (
        <div className="ask-tutor__messages">
          {messages.map((msg, i) => (
            <div key={i} className={`ask-tutor__msg ask-tutor__msg--${msg.role}`}>
              {msg.role === 'user' ? (
                <div className="ask-tutor__user-msg">
                  <p className="ask-tutor__user-query">{msg.query}</p>
                  {msg.intent && (
                    <span className="ask-tutor__user-intent">
                      {INTENTS.find((it) => it.id === msg.intent)?.label || msg.intent}
                    </span>
                  )}
                </div>
              ) : msg.response ? (
                <AskResponseCard response={msg.response} />
              ) : null}
            </div>
          ))}
          {loading && (
            <div className="ask-tutor__msg ask-tutor__msg--assistant">
              <div className="ask-tutor__loading">
                <span className="ask-tutor__loading-dot" />
                <span className="ask-tutor__loading-dot" />
                <span className="ask-tutor__loading-dot" />
              </div>
            </div>
          )}
        </div>
      )}

      <div className="ask-tutor__input-area">
        <div className="ask-tutor__intents">
          {INTENTS.map((it) => (
            <button
              key={it.id}
              className={`ask-tutor__intent ${selectedIntent === it.id ? 'ask-tutor__intent--active' : ''}`}
              onClick={() => setSelectedIntent(selectedIntent === it.id ? null : it.id)}
              title={it.desc}
            >
              {it.label}
            </button>
          ))}
        </div>

        <button
          className="ask-tutor__context-toggle"
          onClick={() => setShowContext(!showContext)}
        >
          {showContext ? 'Hide context' : 'Set your context'} {Object.keys(context).length > 0 && `(${Object.keys(context).length} set)`}
        </button>

        {showContext && (
          <div className="ask-tutor__context-panel">
            {Object.entries(CONTEXTS).map(([key, options]) => (
              <div key={key} className="ask-tutor__context-group">
                <label className="ask-tutor__context-label">{key}</label>
                <select
                  className="ask-tutor__context-select"
                  value={context[key] || ''}
                  onChange={(e) => setContext((prev) => {
                    if (e.target.value) return { ...prev, [key]: e.target.value };
                    const next = { ...prev };
                    delete next[key];
                    return next;
                  })}
                >
                  <option value="">Any</option>
                  {options.map((opt) => <option key={opt} value={opt.toLowerCase()}>{opt}</option>)}
                </select>
              </div>
            ))}
          </div>
        )}

        <div className="ask-tutor__query-row">
          <textarea
            className="ask-tutor__textarea"
            placeholder="Ask about any practical capability..."
            value={query}
            onChange={(e) => setQuery(e.target.value)}
            rows={2}
            onKeyDown={(e) => {
              if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                handleSubmit();
              }
            }}
          />
          <button
            className="ask-tutor__submit"
            onClick={handleSubmit}
            disabled={loading || !query.trim()}
          >
            {loading ? '...' : 'Ask'}
          </button>
        </div>
      </div>
    </div>
  );
}

function AskResponseCard({ response }: { response: AskResponse }) {
  return (
    <div className="ask-tutor__response">
      <p className="ask-tutor__answer">{response.answer}</p>

      {response.action_now && (
        <div className="ask-tutor__action-now">
          <h4 className="ask-tutor__section-label">Do this now</h4>
          <p>{response.action_now}</p>
        </div>
      )}

      {response.implementation_plan.length > 0 && (
        <div className="ask-tutor__plan">
          <h4 className="ask-tutor__section-label">Implementation plan</h4>
          <ol className="ask-tutor__steps">
            {response.implementation_plan.map((step) => (
              <StepCard key={step.step} step={step} />
            ))}
          </ol>
        </div>
      )}

      {(response.safety.warnings.length > 0 ||
        response.safety.stop_conditions.length > 0 ||
        response.safety.when_to_get_help.length > 0) && (
        <SafetySection safety={response.safety} />
      )}

      {response.citations.length > 0 && (
        <div className="ask-tutor__citations">
          <h4 className="ask-tutor__section-label">Sources</h4>
          {response.citations.map((cit, i) => (
            <a key={i} href={`/explore/${cit.slug}`} className="ask-tutor__citation">
              <span className="ask-tutor__citation-num">{i + 1}</span>
              <div>
                <strong>{cit.source_title}</strong>
                <p className="ask-tutor__citation-excerpt">"{cit.excerpt}"</p>
                <span className="ask-tutor__citation-strength">{cit.support_strength}</span>
              </div>
            </a>
          ))}
        </div>
      )}

      {response.related_objects.length > 0 && (
        <div className="ask-tutor__related">
          <h4 className="ask-tutor__section-label">Related capabilities</h4>
          <div className="ask-tutor__related-list">
            {response.related_objects.map((obj) => (
              <a key={obj.slug} href={`/explore/${obj.slug}`} className="ask-tutor__related-card">
                <strong>{obj.title}</strong>
                <span className="ask-tutor__related-role">{obj.role}</span>
              </a>
            ))}
          </div>
        </div>
      )}

      {response.uncertainties.length > 0 && (
        <div className="ask-tutor__gaps">
          <h4 className="ask-tutor__section-label">Gaps & uncertainties</h4>
          <ul>
            {response.uncertainties.map((gap, i) => <li key={i}>{gap}</li>)}
          </ul>
        </div>
      )}
    </div>
  );
}

function StepCard({ step }: { step: ImplementationStep }) {
  return (
    <li className="ask-tutor__step">
      <div className="ask-tutor__step-action">{step.action}</div>
      {(step.tools.length > 0 || step.materials.length > 0) && (
        <div className="ask-tutor__step-needs">
          {step.tools.length > 0 && (
            <span className="ask-tutor__step-tools">Tools: {step.tools.join(', ')}</span>
          )}
          {step.materials.length > 0 && (
            <span className="ask-tutor__step-materials">Materials: {step.materials.join(', ')}</span>
          )}
        </div>
      )}
      {step.time_estimate && (
        <span className="ask-tutor__step-time">{step.time_estimate}</span>
      )}
      {step.source_slug && (
        <a href={`/explore/${step.source_slug}`} className="ask-tutor__step-source">Source</a>
      )}
    </li>
  );
}

function SafetySection({ safety }: { safety: SafetyBlock }) {
  return (
    <div className="ask-tutor__safety">
      <h4 className="ask-tutor__section-label">Safety</h4>
      {safety.warnings.length > 0 && (
        <div className="ask-tutor__safety-group ask-tutor__safety-group--warnings">
          <strong>Warnings</strong>
          <ul>{safety.warnings.map((w, i) => <li key={i}>{w}</li>)}</ul>
        </div>
      )}
      {safety.stop_conditions.length > 0 && (
        <div className="ask-tutor__safety-group ask-tutor__safety-group--stop">
          <strong>Stop if</strong>
          <ul>{safety.stop_conditions.map((s, i) => <li key={i}>{s}</li>)}</ul>
        </div>
      )}
      {safety.when_to_get_help.length > 0 && (
        <div className="ask-tutor__safety-group ask-tutor__safety-group--help">
          <strong>Get help when</strong>
          <ul>{safety.when_to_get_help.map((h, i) => <li key={i}>{h}</li>)}</ul>
        </div>
      )}
    </div>
  );
}

function generateMockResponse(query: string, intent: RetrievalIntent, nodes: GraphNode[]): AskResponse {
  const q = query.toLowerCase();
  const relevant = nodes
    .filter((n) => n.title.toLowerCase().includes(q) || n.plain_language.toLowerCase().includes(q) || n.domain.toLowerCase().includes(q))
    .slice(0, 3);

  if (relevant.length === 0) {
    const fallback = nodes.slice(0, 2);
    relevant.push(...fallback);
  }

  const intentAnswers: Record<string, string> = {
    how_to: `Here's how to approach "${query}". Start with the foundational concepts, then move to the practical skills. The implementation plan below breaks this into concrete steps.`,
    learn_path: `For learning about "${query}", follow the implementation plan below in order. Each step builds on the previous one.`,
    why: `Understanding "${query}" matters because practical capability in this area directly affects household resilience.`,
    compare_options: `There are several approaches to "${query}". Review the related capabilities below to compare trade-offs for your situation.`,
    localize: `Adapting "${query}" to your specific situation depends on your housing type, climate, budget, and experience level.`,
    debug_failure: `If something went wrong with "${query}", check the safety section below for common failure modes.`,
    teach_forward: `To teach someone else about "${query}", start with the plain-language explanation and use the implementation plan as a teaching outline.`,
    safety_check: `Safety assessment for "${query}": check the safety section below. Objects marked "expert only" require professional guidance.`,
    what_changed: `For recent updates on "${query}", check the related capabilities below.`,
  };

  return {
    answer: intentAnswers[intent] || intentAnswers.how_to,
    action_now: relevant.length > 0 ? `Start by reading "${relevant[0].title}" to understand the fundamentals.` : null,
    implementation_plan: relevant.slice(0, 3).map((n, i) => ({
      step: i + 1,
      action: `Review ${n.title} and understand the core ${n.type === 'concept_note' ? 'concept' : 'technique'}`,
      tools: [],
      materials: [],
      time_estimate: '15-30 minutes',
      source_slug: n.slug,
    })),
    safety: {
      warnings: [],
      stop_conditions: [],
      when_to_get_help: relevant.some((n) => n.risk_band === 'high' || n.risk_band === 'expert_only')
        ? ['If you encounter any situation involving high-voltage electrical work or structural modifications']
        : [],
    },
    citations: relevant.slice(0, 2).map((n) => ({
      source_title: n.title,
      slug: n.slug,
      excerpt: n.plain_language.length > 120 ? n.plain_language.substring(0, 117) + '...' : n.plain_language,
      page_range: null,
      support_strength: 'moderate',
    })),
    related_objects: relevant.map((n) => ({
      slug: n.slug,
      title: n.title,
      role: n.type === 'concept_note' ? 'foundational' : 'practical',
    })),
    uncertainties: relevant.length < 2 ? ['Limited coverage in the current graph for this specific topic.'] : [],
    resolved_intent: intent,
    conversation_id: null,
    retrieval_run_id: null,
  };
}
```
